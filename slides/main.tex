%% ============================================================
\section{Актуальность}
%% ============================================================

\begin{frame}{Актуальность}
  \begin{itemize}
    \item Молочное животноводство --- одна из ключевых отраслей сельского хозяйства
    \item Рост поголовья требует автоматизации учёта:
      \begin{itemize}
        \item Данные о животных, надоях, кормлении, воспроизводстве
        \item Интеграция с роботизированными системами доения (Lely)
      \end{itemize}
    \item Существующие решения:
      \begin{itemize}
        \item Коммерческие --- дорогие, закрытые, сложно адаптируемые
        \item Электронные таблицы --- ручной ввод, отсутствие аналитики
      \end{itemize}
    \item Необходимо: доступная, расширяемая,-self-hosted система
  \end{itemize}
\end{frame}

%% ============================================================
\section{Цель и задачи}
%% ============================================================

\begin{frame}{Цель работы}
  % TODO: уточнить формулировку
  \begin{block}{Цель}
    Разработка веб-приложения для комплексного управления молочной фермой с интеграцией роботизированных систем доения.
  \end{block}

  \vspace{1em}
  \textbf{Задачи:}
  \begin{enumerate}
    \item Спроектировать архитектуру и базу данных
    \item Реализовать REST API на Rust/Axum
    \item Разработать SPA-интерфейс на SvelteKit 5
    \item Обеспечить интеграцию с Lely Cloud API
    \item Реализовать аналитику и генерацию отчётов
    \item Настроить контейнеризацию и развёртывание
  \end{enumerate}
\end{frame}

        \item Fitness, Grazing
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Библиотека}
      \begin{itemize}
        \item \textbf{14 API модулей} --- REST клиент
        \item \textbf{16 UI компонентов}:
          \begin{itemize}
            \item DataTable (sort, pagination, responsive)
            \item Modal, ConfirmDialog, FormField
            \item Toaster (4 типа + details)
            \item TabBar (ARIA accessible)
          \end{itemize}
        \item \textbf{Stores}: auth, theme, toast, preferences
        \item \textbf{Utils}: validators, useCrudModal, usePaginatedList
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Frontend: SvelteKit 5 (runes)}
  \begin{itemize}
    \item \textbf{\$state} --- реактивное состояние компонентов
    \item \textbf{\$derived} --- вычисляемые значения (фильтрация, сортировка)
    \item \textbf{\$effect} --- побочные эффекты (fetch, subscriptions)
    \item \textbf{\$props} --- типобезопасные входные параметры
    \item \textbf{SSR} --- load functions (\texttt{+page.ts}) для 15 маршрутов
    \item \textbf{hooks.server.ts} --- JWT decode, auth guard, redirect
  \end{itemize}

  \vspace{1em}
  \textbf{Ключевые паттерны:}
  \begin{itemize}
    \item \texttt{useCrudModal()} --- универсальный CRUD с формой
    \item \texttt{usePaginatedList()} --- загрузка с пагинацией
    \item Responsive tables --- скрытие столбцов на мобильных
  \end{itemize}
\end{frame}
